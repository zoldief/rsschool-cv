function persistence(num) {
    const reducer = (a, b) => a * b;
    num = num.toString();
    num = num.split('');
    let k = 0;
    while (num.length !== 1) {
        num = num.reduce(reducer);
        num = num.toString();
        k++;
        num = num.split('');
    }
    return k
}
